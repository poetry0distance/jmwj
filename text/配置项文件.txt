public enum CarrierCodeEnum {

    CHINA_MOBILE_1("46000", "ChinaMobile"),
    CHINA_MOBILE_2("46002", "ChinaMobile"),
    CHINA_UNICOM("46001", "ChinaUnicom"),
    CHINA_TELECOM("46003", "ChinaTelecom");

    private String carrierCode;
    private String carrierName;


    CarrierCodeEnum(String carrierCode, String carrierName) {
        this.carrierName = carrierName;
        this.carrierCode = carrierCode;
    }

    public String getCarrierName() {
        return carrierName;
    }

    private static final Map<String,CarrierCodeEnum> map = new HashMap<>();

    static {
        Arrays.stream(CarrierCodeEnum.values()).forEach(carrier-> map.put(carrier.carrierCode, carrier));
    }

    public static CarrierCodeEnum getByCode(String code){
        return map.get(code);
    }
}
